import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from geometry_msgs.msg import PointStamped
from visualization_msgs.msg import Marker
from cv_bridge import CvBridge
from ultralytics import YOLO
import tf2_ros
import tf2_geometry_msgs
import cv2
import time


class YoloMapMarkerNode(Node):
    def __init__(self):
        super().__init__('yolo_map_marker_node')

        self.model = YOLO('/home/robot/runs/runs/detect/train3/weights/best.pt')
        self.bridge = CvBridge()

        self.create_subscription(Image, '/camera/color/image_raw', self.image_callback, 10)
        self.marker_pub = self.create_publisher(Marker, '/badminton_marker', 10)

        self.tf_buffer = tf2_ros.Buffer()
        self.tf_listener = tf2_ros.TransformListener(self.tf_buffer, self)

        self.marker_id_counter = 0

        # 改成參數設定 camera frame，預設用 camera_color_optical_frame
        self.declare_parameter('camera_frame', 'camera_color_optical_frame')
        self.camera_frame = self.get_parameter('camera_frame').get_parameter_value().string_value

        self.get_logger().info(f'YOLO map marker node started, using camera frame: {self.camera_frame}')

    def image_callback(self, msg):
        try:
            frame = self.bridge.imgmsg_to_cv2(msg, 'bgr8')
        except Exception as e:
            self.get_logger().error(f"CvBridge conversion failed: {e}")
            return

        results = self.model(frame)
        height, width, _ = frame.shape

        # 重置 id，避免持續累積導致問題
        self.marker_id_counter = 0

        for r in results:
            for box in r.boxes:
                cls = int(box.cls[0])
                label = self.model.names[cls] if hasattr(self.model, 'names') else str(cls)
                if label.lower() != "badminton":
                    continue

                x1, y1, x2, y2 = map(int, box.xyxy[0])
                cx = (x1 + x2) // 2
                cy = (y1 + y2) // 2

                # 繪製框框
                cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
                cv2.putText(frame, f"{label}", (x1, y1 - 10),
                            cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)

                # 固定深度 0.3m 計算相機座標
                z = 0.3
                fx = 600.0
                fy = 600.0
                cx_cam = width / 2
                cy_cam = height / 2

                x = (cx - cx_cam) * z / fx
                y = (cy - cy_cam) * z / fy

                camera_point = PointStamped()
                camera_point.header.frame_id = self.camera_frame
                camera_point.header.stamp = self.get_clock().now().to_msg()
                camera_point.point.x = x
                camera_point.point.y = y
                camera_point.point.z = z

                try:
                    transform = self.tf_buffer.lookup_transform(
                        'map',
                        self.camera_frame,
                        rclpy.time.Time()
                    )
                    map_point = tf2_geometry_msgs.do_transform_point(camera_point, transform)

                    # Clamp Z to 0 (floor level)
                    if map_point.point.z < 0:
                        map_point.point.z = 0.0

                    self.publish_marker(map_point.point)

                    self.get_logger().info(
                        f"Marked badminton at map: ({map_point.point.x:.2f}, {map_point.point.y:.2f}, {map_point.point.z:.2f})"
                    )
                except Exception as e:
                    self.get_logger().warn(f"TF transform failed: {e}")

        try:
            cv2.imshow("YOLO Marker (Fixed Depth)", frame)
            cv2.waitKey(10)
        except Exception as e:
            self.get_logger().warn(f"GUI error: {e}")

    def publish_marker(self, point):
        point.x = point.z + 0.45
        #if point.z < 0: z+=0.16
        point.z = 0.05  # 防止跑到地板以下
        marker = Marker()
        marker.header.frame_id = "map"
        marker.header.stamp = self.get_clock().now().to_msg()
        marker.ns = "badminton"
        marker.id = self.marker_id_counter
        self.marker_id_counter += 1

        marker.type = Marker.SPHERE
        marker.action = Marker.ADD
        marker.pose.position = point
        marker.pose.orientation.w = 1.0
        marker.scale.x = 0.05
        marker.scale.y = 0.05
        marker.scale.z = 0.05
        marker.color.r = 1.0
        marker.color.g = 0.5
        marker.color.b = 0.0
        marker.color.a = 1.0
        marker.lifetime.sec = 5  # 5秒後自動消失

        self.marker_pub.publish(marker)

    def destroy_node(self):
        cv2.destroyAllWindows()
        super().destroy_node()


def main(args=None):
    rclpy.init(args=args)
    node = YoloMapMarkerNode()
    try:
        rclpy.spin(node)
    except KeyboardInterrupt:
        pass
    finally:
        node.destroy_node()
        rclpy.shutdown()


if __name__ == '__main__':
    main()
